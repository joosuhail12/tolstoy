# List Webhooks

<Badge variant="success">GET</Badge> `/webhooks`

Retrieve a paginated list of webhook endpoints configured in your organization. Webhooks allow external systems to trigger flows and send data to Tolstoy.

## Request

### **HTTP Request**

```http
GET https://api.tolstoy.dev/webhooks
```

### **Query Parameters**

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `limit` | integer | 20 | Number of webhooks per page (1-100) |
| `cursor` | string | - | Pagination cursor for next page |
| `sort` | string | `created_at` | Sort field (`name`, `created_at`, `updated_at`, `last_triggered`) |
| `order` | string | `desc` | Sort order (`asc` or `desc`) |
| `search` | string | - | Search webhooks by name or URL |
| `status` | string | - | Filter by status (`active`, `inactive`, `paused`) |
| `flow_id` | string | - | Filter by associated flow |
| `created_after` | string | - | Filter by creation date (ISO 8601) |
| `created_before` | string | - | Filter by creation date (ISO 8601) |

### **Webhook Object**

| Field | Type | Description |
|-------|------|-------------|
| `id` | string | Unique webhook identifier |
| `name` | string | Webhook name (used in API references) |
| `display_name` | string | Human-readable webhook name |
| `description` | string | Detailed webhook description |
| `url` | string | Webhook endpoint URL |
| `status` | string | Webhook status (`active`, `inactive`, `paused`) |
| `flow_id` | string | Associated flow identifier |
| `flow_name` | string | Associated flow display name |
| `config` | object | Webhook configuration settings |
| `validation` | object | Request validation rules |
| `security` | object | Security and access controls |
| `statistics` | object | Usage and performance statistics |
| `tags` | array | Tags for organization |
| `created_by` | string | User who created the webhook |
| `created_at` | string | ISO 8601 timestamp of creation |
| `updated_at` | string | ISO 8601 timestamp of last update |

<CodeGroup>

```javascript Webhook Analytics
async function analyzeWebhookPerformance(apiKey) {
  const response = await fetch('https://api.tolstoy.dev/webhooks?limit=100', {
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json'
    }
  });
  
  const { webhooks } = await response.json();
  
  console.log('Webhook Performance Analysis');
  console.log('============================');
  
  // Overall statistics
  const totalRequests = webhooks.reduce((sum, w) => sum + w.statistics.total_requests, 0);
  const totalSuccessful = webhooks.reduce((sum, w) => sum + w.statistics.successful_requests, 0);
  const overallSuccessRate = totalRequests > 0 ? totalSuccessful / totalRequests : 0;
  
  console.log(`Total webhooks: ${webhooks.length}`);
  console.log(`Total requests: ${totalRequests.toLocaleString()}`);
  console.log(`Overall success rate: ${(overallSuccessRate * 100).toFixed(2)}%`);
  console.log('');
  
  // Performance ranking
  const sortedByPerformance = [...webhooks].sort((a, b) => {
    // Sort by success rate, then by request volume
    const aScore = a.statistics.success_rate * 0.7 + 
                   (a.statistics.requests_last_24h / 1000) * 0.3;
    const bScore = b.statistics.success_rate * 0.7 + 
                   (b.statistics.requests_last_24h / 1000) * 0.3;
    return bScore - aScore;
  });
  
  console.log('Top Performing Webhooks:');
  sortedByPerformance.slice(0, 5).forEach((webhook, index) => {
    console.log(`${index + 1}. ${webhook.display_name}`);
    console.log(`   Success rate: ${(webhook.statistics.success_rate * 100).toFixed(2)}%`);
    console.log(`   Avg response: ${webhook.statistics.avg_response_time_ms}ms`);
    console.log(`   24h requests: ${webhook.statistics.requests_last_24h}`);
  });
  
  // Problem detection
  const problemWebhooks = webhooks.filter(w => 
    w.statistics.success_rate < 0.95 || 
    w.statistics.avg_response_time_ms > 1000 ||
    w.status !== 'active'
  );
  
  if (problemWebhooks.length > 0) {
    console.log('\nWebhooks Needing Attention:');
    problemWebhooks.forEach(webhook => {
      const issues = [];
      if (webhook.statistics.success_rate < 0.95) {
        issues.push(`Low success rate: ${(webhook.statistics.success_rate * 100).toFixed(2)}%`);
      }
      if (webhook.statistics.avg_response_time_ms > 1000) {
        issues.push(`Slow response: ${webhook.statistics.avg_response_time_ms}ms`);
      }
      if (webhook.status !== 'active') {
        issues.push(`Status: ${webhook.status}`);
      }
      
      console.log(`âš ï¸  ${webhook.display_name}`);
      issues.forEach(issue => console.log(`   - ${issue}`));
      
      // Show top error types
      const errors = webhook.statistics.error_breakdown;
      const topErrors = Object.entries(errors)
        .sort(([,a], [,b]) => b - a)
        .slice(0, 2);
      
      if (topErrors.length > 0) {
        console.log(`   Top errors: ${topErrors.map(([type, count]) => `${type}(${count})`).join(', ')}`);
      }
    });
  }
  
  return {
    total: webhooks.length,
    totalRequests,
    overallSuccessRate,
    topPerforming: sortedByPerformance.slice(0, 3),
    problemWebhooks
  };
}

// Usage
const analytics = await analyzeWebhookPerformance(apiKey);
```

### **Security Audit**

<CodeGroup>

```javascript Security Audit
async function auditWebhookSecurity(apiKey) {
  const response = await fetch('https://api.tolstoy.dev/webhooks', {
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json'
    }
  });
  
  const { webhooks } = await response.json();
  const securityIssues = [];
  
  webhooks.forEach(webhook => {
    const issues = [];
    
    // Check HTTPS requirement
    if (!webhook.security.https_required) {
      issues.push('HTTPS not required');
    }
    
    // Check authentication
    if (!webhook.config.authentication) {
      issues.push('No authentication configured');
    } else if (webhook.config.authentication.type === 'none') {
      issues.push('Authentication disabled');
    }
    
    // Check IP whitelist
    if (!webhook.security.ip_whitelist || webhook.security.ip_whitelist.length === 0) {
      issues.push('No IP restrictions');
    }
    
    // Check rate limits
    if (!webhook.config.rate_limit || webhook.config.rate_limit.requests_per_minute > 1000) {
      issues.push('High/no rate limit');
    }
    
    if (issues.length > 0) {
      securityIssues.push({
        webhook: webhook.display_name,
        url: webhook.url,
        issues
      });
    }
  });
  
  console.log('Webhook Security Audit');
  console.log('======================');
  
  if (securityIssues.length === 0) {
    console.log('âœ… All webhooks pass security checks');
  } else {
    console.log(`âš ï¸  Found ${securityIssues.length} webhooks with security issues:`);
    securityIssues.forEach(item => {
      console.log(`\nðŸ”´ ${item.webhook}`);
      item.issues.forEach(issue => console.log(`   - ${issue}`));
    });
  }
  
  return securityIssues;
}

// Usage
const securityReport = await auditWebhookSecurity(apiKey);
```
