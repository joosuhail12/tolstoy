# List Actions

Retrieve a comprehensive list of available actions that can be used in workflow steps. Actions are pre-built integrations and operations that perform specific tasks within your flows.

<RequestExample>

```bash cURL
curl -X GET "https://tolstoy.getpullse.com/actions?category=communication&search=slack&limit=20" \
  -H "x-org-id: org_1234567890abcdef" \
  -H "x-user-id: user_abcdef1234567890" \
  -H "Content-Type: application/json"
```

```typescript TypeScript
interface ListActionsParams {
  category?: string;
  search?: string;
  limit?: number;
  cursor?: string;
  sort_by?: 'name' | 'category' | 'popularity' | 'created_at';
  sort_order?: 'asc' | 'desc';
}

const listActions = async (params: ListActionsParams = {}) => {
  const searchParams = new URLSearchParams();
  
  Object.entries(params).forEach(([key, value]) => {
    if (value !== undefined) {
      searchParams.set(key, value.toString());
    }
  });

  const response = await fetch(`https://tolstoy.getpullse.com/actions?${searchParams}`, {
    method: 'GET',
    headers: {
      'x-org-id': 'org_1234567890abcdef',
      'x-user-id': 'user_abcdef1234567890',
      'Content-Type': 'application/json'
    }
  });

  const result = await response.json();
  
  if (!result.success) {
    throw new Error(`Failed to list actions: ${result.error.message}`);
  }

  return result.data;
};

// Get communication actions
const communicationActions = await listActions({
  category: 'communication',
  limit: 50
});

console.log(`Found ${communicationActions.length} communication actions`);
```

```python Python
import requests
from typing import Optional, List, Dict, Any

def list_actions(
    category: Optional[str] = None,
    search: Optional[str] = None,
    limit: int = 20,
    cursor: Optional[str] = None,
    sort_by: str = 'name',
    sort_order: str = 'asc'
) -> List[Dict[str, Any]]:
    """List available actions with filtering and pagination."""
    
    headers = {
        'x-org-id': 'org_1234567890abcdef',
        'x-user-id': 'user_abcdef1234567890',
        'Content-Type': 'application/json'
    }
    
    params = {
        'limit': limit,
        'sort_by': sort_by,
        'sort_order': sort_order
    }
    
    if category:
        params['category'] = category
    if search:
        params['search'] = search
    if cursor:
        params['cursor'] = cursor
    
    response = requests.get(
        'https://tolstoy.getpullse.com/actions',
        headers=headers,
        params=params
    )
    response.raise_for_status()
    
    result = response.json()
    if not result['success']:
        raise Exception(f"Failed to list actions: {result['error']['message']}")
    
    return result['data']

# Get all HTTP actions
http_actions = list_actions(category='http', limit=100)
print(f"Found {len(http_actions)} HTTP actions")

# Search for email actions
email_actions = list_actions(search='email', limit=50)
for action in email_actions:
    print(f"- {action['name']}: {action['description']}")
```

```javascript Node.js
const axios = require('axios');

class ActionsClient {
  constructor(orgId, userId, baseUrl = 'https://tolstoy.getpullse.com') {
    this.orgId = orgId;
    this.userId = userId;
    this.baseUrl = baseUrl;
    this.headers = {
      'x-org-id': orgId,
      'x-user-id': userId,
      'Content-Type': 'application/json'
    };
  }

  async listActions(options = {}) {
    const {
      category,
      search,
      limit = 20,
      cursor,
      sort_by = 'name',
      sort_order = 'asc'
    } = options;

    const params = new URLSearchParams({
      limit: limit.toString(),
      sort_by,
      sort_order
    });

    if (category) params.set('category', category);
    if (search) params.set('search', search);
    if (cursor) params.set('cursor', cursor);

    try {
      const response = await axios.get(`${this.baseUrl}/actions`, {
        headers: this.headers,
        params: Object.fromEntries(params)
      });

      if (!response.data.success) {
        throw new Error(response.data.error.message);
      }

      return response.data.data;
    } catch (error) {
      if (error.response) {
        throw new Error(`API Error: ${error.response.status} - ${error.response.data?.error?.message || error.message}`);
      }
      throw error;
    }
  }

  async getAllActions() {
    const allActions = [];
    let cursor = null;

    do {
      const response = await this.listActions({
        limit: 100,
        cursor
      });

      allActions.push(...response);
      
      // Check if there are more pages (this would come from pagination info)
      cursor = response.pagination?.cursor || null;
    } while (cursor);

    return allActions;
  }

  async getActionsByCategory() {
    const actions = await this.getAllActions();
    const categorized = {};

    actions.forEach(action => {
      const category = action.category;
      if (!categorized[category]) {
        categorized[category] = [];
      }
      categorized[category].push(action);
    });

    return categorized;
  }
}

// Usage
const client = new ActionsClient('org_1234567890abcdef', 'user_abcdef1234567890');

// Get popular actions
const popularActions = await client.listActions({
  sort_by: 'popularity',
  sort_order: 'desc',
  limit: 10
});

console.log('Top 10 most popular actions:');
popularActions.forEach((action, index) => {
  console.log(`${index + 1}. ${action.name} (${action.category})`);
});
```

## Query Parameters

<ParamField query="category" type="string">
Filter actions by category. Available categories:
- `communication` - Email, Slack, Teams, SMS, etc.
- `http` - HTTP requests, API calls, webhooks
- `database` - Database operations and queries
- `data` - Data transformation and manipulation
- `file` - File operations and storage
- `utility` - Utility functions and helpers
- `custom` - Custom user-defined actions
</ParamField>

<ParamField query="search" type="string">
Search actions by name, description, or tags. Supports partial matching and is case-insensitive.
</ParamField>

<ParamField query="limit" type="integer" default="20">
Number of actions to return per page. Maximum value is 100.
</ParamField>

<ParamField query="cursor" type="string">
Pagination cursor for retrieving the next page of results.
</ParamField>

<ParamField query="sort_by" type="string" default="name">
Sort field. Available options:
- `name` - Sort alphabetically by action name
- `category` - Sort by action category
- `popularity` - Sort by usage frequency
- `created_at` - Sort by when action was added
</ParamField>

<ParamField query="sort_order" type="string" default="asc">
Sort direction: `asc` (ascending) or `desc` (descending)
</ParamField>

<ParamField query="tags" type="string">
Filter by tags. Comma-separated list (e.g., `popular,integration,beta`).
</ParamField>

<ParamField query="featured_only" type="boolean" default="false">
When `true`, only return featured/recommended actions.
</ParamField>
