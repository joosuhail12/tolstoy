# Execute Action

Execute a single action independently outside of a flow context. This endpoint is perfect for testing actions, one-off operations, or building custom integrations that don't require full workflow orchestration.

<RequestExample>

```bash cURL
curl -X POST "https://tolstoy.getpullse.com/actions/execute" \
  -H "x-org-id: org_1234567890abcdef" \
  -H "x-user-id: user_abcdef1234567890" \
  -H "Content-Type: application/json" \
  -d '{
    "action": "email-send",
    "inputs": {
      "to": "customer@example.com",
      "subject": "Welcome to our platform!",
      "body": "Thanks for signing up. We are excited to have you on board!",
      "template": "welcome-template"
    },
    "timeout_seconds": 60,
    "tags": ["manual-execution", "customer-onboarding"]
  }'
```

```typescript TypeScript
interface ExecuteActionRequest {
  action: string;
  inputs: Record<string, any>;
  timeout_seconds?: number;
  tags?: string[];
  dry_run?: boolean;
  priority?: 'low' | 'normal' | 'high';
}

interface ExecuteActionResponse {
  success: boolean;
  data: {
    execution_id: string;
    action: string;
    status: 'completed' | 'failed' | 'timeout';
    inputs: Record<string, any>;
    outputs?: Record<string, any>;
    error?: {
      code: string;
      message: string;
      details?: Record<string, any>;
    };
    performance: {
      duration_ms: number;
      memory_used_mb?: number;
    };
    executed_at: string;
  };
}

const executeAction = async (request: ExecuteActionRequest): Promise<ExecuteActionResponse> => {
  const response = await fetch('https://tolstoy.getpullse.com/actions/execute', {
    method: 'POST',
    headers: {
      'x-org-id': 'org_1234567890abcdef',
      'x-user-id': 'user_abcdef1234567890',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(request)
  });

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}: ${response.statusText}`);
  }

  return response.json();
};

// Send welcome email
try {
  const result = await executeAction({
    action: 'email-send',
    inputs: {
      to: 'customer@example.com',
      subject: 'Welcome!',
      template: 'welcome-template',
      variables: {
        customer_name: 'John Doe',
        plan_type: 'Premium'
      }
    },
    timeout_seconds: 60,
    tags: ['welcome-email', 'onboarding']
  });

  if (result.success && result.data.status === 'completed') {
    console.log('Email sent successfully:', result.data.outputs?.message_id);
  } else {
    console.error('Email failed:', result.data.error?.message);
  }
} catch (error) {
  console.error('Request failed:', error);
}
```

```python Python
import requests
import json
from typing import Dict, Any, Optional, List

class ActionExecutor:
    def __init__(self, org_id: str, user_id: str, base_url: str = "https://tolstoy.getpullse.com"):
        self.org_id = org_id
        self.user_id = user_id
        self.base_url = base_url
        self.headers = {
            'x-org-id': org_id,
            'x-user-id': user_id,
            'Content-Type': 'application/json'
        }

    def execute_action(
        self,
        action: str,
        inputs: Dict[str, Any],
        timeout_seconds: int = 30,
        tags: Optional[List[str]] = None,
        dry_run: bool = False,
        priority: str = 'normal'
    ) -> Dict[str, Any]:
        """Execute a single action with the given inputs."""
        
        payload = {
            'action': action,
            'inputs': inputs,
            'timeout_seconds': timeout_seconds,
            'priority': priority
        }
        
        if tags:
            payload['tags'] = tags
        if dry_run:
            payload['dry_run'] = dry_run

        response = requests.post(
            f"{self.base_url}/actions/execute",
            headers=self.headers,
            json=payload
        )
        response.raise_for_status()
        
        return response.json()

    def send_slack_message(self, channel: str, message: str, **kwargs) -> Dict[str, Any]:
        """Helper method to send Slack messages."""
        inputs = {
            'channel': channel,
            'message': message,
            **kwargs
        }
        
        return self.execute_action('slack-message', inputs, tags=['slack', 'notification'])

    def make_http_request(self, url: str, method: str = 'GET', **kwargs) -> Dict[str, Any]:
        """Helper method to make HTTP requests."""
        inputs = {
            'url': url,
            'method': method,
            **kwargs
        }
        
        return self.execute_action('http-request', inputs, tags=['http', 'api-call'])

    def validate_email(self, email: str) -> Dict[str, Any]:
        """Helper method to validate email addresses."""
        return self.execute_action('email-validate', {'email': email}, tags=['validation'])
```

```javascript Node.js
const axios = require('axios');

class ActionExecutor {
  constructor(orgId, userId, baseUrl = 'https://tolstoy.getpullse.com') {
    this.orgId = orgId;
    this.userId = userId;
    this.baseUrl = baseUrl;
    this.headers = {
      'x-org-id': orgId,
      'x-user-id': userId,
      'Content-Type': 'application/json'
    };
  }

  async executeAction(action, inputs, options = {}) {
    const {
      timeout_seconds = 30,
      tags = [],
      dry_run = false,
      priority = 'normal'
    } = options;

    const payload = {
      action,
      inputs,
      timeout_seconds,
      tags,
      priority
    };

    if (dry_run) {
      payload.dry_run = true;
    }

    try {
      const response = await axios.post(
        `${this.baseUrl}/actions/execute`,
        payload,
        { headers: this.headers }
      );

      return response.data;
    } catch (error) {
      if (error.response) {
        throw new Error(`API Error: ${error.response.status} - ${error.response.data?.error?.message || error.message}`);
      }
      throw error;
    }
  }

  // Helper methods for common actions
  async sendEmail(to, subject, body, options = {}) {
    return this.executeAction('email-send', {
      to,
      subject,
      body,
      ...options
    }, { tags: ['email', 'notification'] });
  }

  async sendSlackMessage(channel, message, options = {}) {
    return this.executeAction('slack-message', {
      channel,
      message,
      ...options
    }, { tags: ['slack', 'notification'] });
  }

  async makeHttpRequest(url, method = 'GET', options = {}) {
    return this.executeAction('http-request', {
      url,
      method,
      ...options
    }, { tags: ['http', 'api-call'] });
  }

  async transformData(data, transformRules) {
    return this.executeAction('data-transform', {
      data,
      rules: transformRules
    }, { tags: ['data', 'transformation'] });
  }

  async validateData(data, schema) {
    return this.executeAction('data-validate', {
      data,
      schema
    }, { tags: ['validation'] });
  }
}

// Usage examples
const executor = new ActionExecutor('org_1234567890abcdef', 'user_abcdef1234567890');

async function processNewCustomer(customerData) {
  try {
    // Validate customer data
    const validation = await executor.validateData(customerData, {
      type: 'object',
      required: ['email', 'name'],
      properties: {
        email: { type: 'string', format: 'email' },
        name: { type: 'string', minLength: 1 }
      }
    });

    if (validation.data.status !== 'completed' || !validation.data.outputs.valid) {
      throw new Error('Customer data validation failed');
    }

    // Send welcome email
    const emailResult = await executor.sendEmail(
      customerData.email,
      'Welcome to our platform!',
      null, // Will use template instead of body
      {
        template: 'welcome-template',
        variables: {
          customer_name: customerData.name,
          login_url: 'https://app.example.com/login'
        }
      }
    );

    // Notify team on Slack
    const slackResult = await executor.sendSlackMessage(
      '#new-customers',
      `ðŸŽ‰ New customer signed up: ${customerData.name}`,
      {
        attachments: [{
          color: 'good',
          fields: [
            { title: 'Name', value: customerData.name, short: true },
            { title: 'Email', value: customerData.email, short: true }
          ]
        }]
      }
    );

    // Update CRM
    const crmResult = await executor.makeHttpRequest(
      'https://api.crm.com/contacts',
      'POST',
      {
        headers: { 'Authorization': 'Bearer crm-token' },
        body: {
          email: customerData.email,
          name: customerData.name,
          source: 'website_signup',
          tags: ['new_customer']
        }
      }
    );

    console.log('Customer onboarding completed successfully');
    return {
      email_sent: emailResult.success,
      slack_notified: slackResult.success,
      crm_updated: crmResult.success
    };

  } catch (error) {
    console.error('Customer onboarding failed:', error.message);
    throw error;
  }
}

// Process a new customer
processNewCustomer({
  name: 'Jane Smith',
  email: 'jane@example.com',
  plan: 'premium'
}).then(result => {
  console.log('Onboarding result:', result);
}).catch(error => {
  console.error('Error:', error.message);
});
```

## Request Body

<ParamField body="action" type="string" required>
The action identifier to execute. Must match an action available in your organization.

**Format:** Action IDs are kebab-case strings (e.g., `email-send`, `slack-message`, `http-request`)

**Examples:**
- `email-send` - Send email messages
- `slack-message` - Send Slack notifications  
- `http-request` - Make HTTP API calls
- `data-transform` - Transform data structures
</ParamField>

<ParamField body="inputs" type="object" required>
Input parameters for the action execution. Must conform to the action's input schema.

**Format:** Key-value pairs where keys match the action's expected input parameter names.

**Validation:** Inputs are validated against the action's JSON schema before execution.
</ParamField>

<ParamField body="timeout_seconds" type="integer" default="30">
Maximum execution time in seconds before the action times out.

**Constraints:**
- Minimum: 1 second
- Maximum: 300 seconds (5 minutes)
- Different actions may have stricter limits
</ParamField>

<ParamField body="tags" type="string[]">
Optional tags to categorize this execution for tracking and analytics.

**Constraints:**
- Maximum 10 tags per execution
- Each tag: 1-50 characters
- Tags are case-insensitive
</ParamField>

<ParamField body="dry_run" type="boolean" default="false">
When true, validates inputs and returns execution plan without actually running the action.

**Use cases:**
- Testing action configurations
- Validating input data
- Debugging integration issues
</ParamField>

<ParamField body="priority" type="string" default="normal">
Execution priority level for queue management.

**Available priorities:**
- `low` - Lower priority (may wait longer in queue)
- `normal` - Standard priority
- `high` - Higher priority (faster processing)
</ParamField>

<ParamField body="context" type="object">
Optional execution context for advanced scenarios.

<Expandable title="Context Fields">
  <ParamField body="user_context" type="object">
    User-specific context data available to the action
  </ParamField>
  <ParamField body="organization_context" type="object">
    Organization-specific context data
  </ParamField>
  <ParamField body="trace_id" type="string">
    Custom trace ID for request correlation
  </ParamField>
</Expandable>
</ParamField>

## Response Fields

<ResponseField name="execution_id" type="string">
Unique identifier for this action execution
</ResponseField>

<ResponseField name="action" type="string">
The action that was executed
</ResponseField>

<ResponseField name="status" type="string">
Execution status:
- `completed` - Action executed successfully
- `failed` - Action failed with error
- `timeout` - Action exceeded timeout limit
</ResponseField>

<ResponseField name="inputs" type="object">
The input parameters that were provided for execution
</ResponseField>

<ResponseField name="outputs" type="object">
Results returned by the action (only present when status is `completed`)
</ResponseField>

    <ResponseField name="details" type="object">Additional error context and debugging information</ResponseField>

</ResponseField>

## Status Codes

<ResponseField name="200" type="Success">
Action executed successfully (regardless of action success/failure)
</ResponseField>

<ResponseField name="400" type="Bad Request">
Invalid request format or missing required fields
</ResponseField>

<ResponseField name="401" type="Unauthorized">
Missing or invalid authentication headers
</ResponseField>

<ResponseField name="403" type="Forbidden">
Insufficient permissions to execute actions
</ResponseField>

<ResponseField name="404" type="Not Found">
Specified action not found or not available
</ResponseField>

<ResponseField name="422" type="Unprocessable Entity">
Action input validation failed
</ResponseField>

## Error Handling

### Common Error Patterns

<Accordion title="Input Validation Errors">
These occur when the provided inputs don't match the action's schema:

```json
{
  "code": "VALIDATION_ERROR",
  "message": "Input validation failed",
  "details": {
    "field": "email",
    "value": "invalid-email",
    "reason": "Must be a valid email address"
  }
}
```

**Resolution:** Check action documentation for correct input format and validate data before sending.
</Accordion>

<Accordion title="Authentication Errors">
These occur when the action requires authentication that isn't properly configured:

```json
{
  "code": "AUTH_ERROR",
  "message": "Authentication failed for action provider",
  "details": {
    "provider": "slack",
    "auth_type": "oauth2",
    "error": "Invalid or expired token"
  }
}
```

**Resolution:** Check your integrations and refresh authentication tokens if needed.
</Accordion>

<Accordion title="Rate Limit Errors">
These occur when you exceed the action's rate limits:

```json
{
  "code": "RATE_LIMITED",
  "message": "Action rate limit exceeded",
  "details": {
    "limit": 100,
    "period": "minute", 
    "retry_after": 45
  }
}
```

**Resolution:** Implement exponential backoff and respect the retry_after value.
</Accordion>

<Accordion title="Provider Errors">
These are errors from external services that the action integrates with:

```json
{
  "code": "PROVIDER_ERROR", 
  "message": "External service returned an error",
  "details": {
    "provider": "sendgrid",
    "provider_code": "400",
    "provider_message": "The from email does not match a verified Sender Identity"
  }
}
```

**Resolution:** Check the provider's documentation and fix the configuration issue.
</Accordion>

## Best Practices

### Input Handling

<Accordion title="Data Validation">
- Always validate inputs client-side before sending requests
- Use dry_run mode to test configurations before production use
- Handle edge cases like null values, empty strings, and special characters
- Consider data size limits (typically 10MB max per execution)
</Accordion>

<Accordion title="Sensitive Data">
- Never include secrets or credentials directly in inputs
- Use secret management features or environment variables
- Be careful with logging and error messages that might expose sensitive data
- Consider using data masking for PII in logs
</Accordion>

### Performance Optimization

<Accordion title="Timeout Management">
- Set appropriate timeouts based on expected action duration
- Consider network latency for HTTP-based actions
- Use shorter timeouts for time-sensitive operations
- Monitor timeout patterns to optimize default values
</Accordion>

<Accordion title="Error Recovery">
- Implement retry logic for transient failures
- Use exponential backoff for rate-limited actions
- Have fallback strategies for critical operations
- Monitor error rates to identify systemic issues
</Accordion>

### Testing and Debugging

<Accordion title="Dry Run Testing">
- Use dry_run mode to validate configurations
- Test with various input scenarios
- Verify authentication and permissions
- Check estimated costs and resource usage
</Accordion>

<Accordion title="Monitoring">
- Use tags consistently for execution tracking
- Monitor performance metrics and success rates
- Set up alerts for high error rates
- Review execution logs regularly
</Accordion>
