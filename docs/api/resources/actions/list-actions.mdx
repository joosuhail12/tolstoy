# List Actions

Retrieve a comprehensive list of all actions available in your organization, including built-in actions, custom actions, and third-party integrations with advanced filtering and search capabilities.

## Endpoint

<CodeGroup>
```http HTTP
GET /actions
```

```bash cURL
curl -X GET "https://tolstoy.getpullse.com/actions" \
  -H "x-org-id: org_1234567890abcdef" \
  -H "x-user-id: user_abcdef1234567890"
```
</CodeGroup>

## Query Parameters

### Pagination

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `page` | number | 1 | Page number for pagination |
| `limit` | number | 50 | Number of actions per page (max 100) |
| `cursor` | string | - | Cursor-based pagination token |

### Filtering

| Parameter | Type | Description |
|-----------|------|-------------|
| `category` | string | Filter by action category |
| `type` | string | Filter by action type (`builtin`, `custom`, `integration`) |
| `status` | string | Filter by status (`active`, `inactive`, `deprecated`) |
| `tags` | string | Comma-separated list of tags to filter by |
| `created_by` | string | Filter by creator user ID |
| `provider` | string | Filter by integration provider (e.g., `slack`, `salesforce`) |

### Search

| Parameter | Type | Description |
|-----------|------|-------------|
| `search` | string | Search in action name, description, and keywords |
| `search_fields` | string | Specific fields to search (`name`, `description`, `keywords`) |

### Sorting

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `sort_by` | string | `name` | Sort field (`name`, `created_at`, `updated_at`, `usage_count`) |
| `sort_order` | string | `asc` | Sort order (`asc` or `desc`) |

```python Python
import tolstoy

client = tolstoy.Client(
    org_id='org_1234567890abcdef',
    user_id='user_abcdef1234567890'

def analyze_available_actions():
    try:
        # Get all actions with statistics
        actions = client.actions.list({
            'limit': 100,
            'sort_by': 'success_rate',
            'sort_order': 'desc'
        })
        
        print(f"Total actions available: {actions['pagination']['total_count']}")
        
        # Analyze by category
        categories = actions['summary']['by_category']
        print("\nðŸ“Š Actions by Category:")
        for category, count in sorted(categories.items(), key=lambda x: x[1], reverse=True):
            print(f"  {category.title()}: {count}")
        
        # Find best performing actions
        best_actions = [
            action for action in actions['actions']
            if action['usage_stats']['success_rate'] > 0.95 
            and action['usage_stats']['total_executions'] > 100
        ]
        
        print(f"\nâœ… High-reliability actions: {len(best_actions)}")
        
        # Display integration providers
        providers = {}
        for action in actions['actions']:
            if action['type'] == 'integration':
                provider = action['provider']['name']
                providers[provider] = providers.get(provider, 0) + 1
        
        print("\nðŸ”— Integration Providers:")
        for provider, count in sorted(providers.items(), key=lambda x: x[1], reverse=True):
            print(f"  {provider.title()}: {count} actions")
        
        return actions
        
    except Exception as error:
        print(f"Failed to analyze actions: {error}")
        raise error
```

```typescript TypeScript - Advanced Search
async function findActionsForUseCase(useCase: string) {
  try {
    // Search for actions related to the use case
    const searchResults = await client.actions.list({
      search: useCase,
      search_fields: 'name,description,keywords',
      status: 'active',
      limit: 20
    });
    
    // Also get actions from relevant categories
    const categoryMap = {
      'email': 'communication',
      'data': 'data',
      'api': 'http',
      'validation': 'validation',
      'notification': 'communication'
    };
    
    const category = categoryMap[useCase.toLowerCase()];
    let categoryResults = [];
    
    if (category) {
      categoryResults = await client.actions.list({
        category,
        status: 'active',
        sort_by: 'usage_count',
        sort_order: 'desc',
        limit: 10
      });
    }
    
    // Combine and deduplicate results
    const allActions = [...searchResults.actions];
    const seenIds = new Set(searchResults.actions.map(a => a.id));
    
    if (categoryResults.actions) {
      categoryResults.actions.forEach(action => {
        if (!seenIds.has(action.id)) {
          allActions.push(action);
        }
      });
    }
    
    // Sort by relevance (combination of usage and success rate)
    const sortedActions = allActions.sort((a, b) => {
      const scoreA = a.usage_stats.success_rate * Math.log(a.usage_stats.total_executions + 1);
      const scoreB = b.usage_stats.success_rate * Math.log(b.usage_stats.total_executions + 1);
      return scoreB - scoreA;
    });
    
    console.log(`Found ${sortedActions.length} actions for "${useCase}"`);
    
    // Display recommendations
    sortedActions.slice(0, 5).forEach((action, i) => {
      const score = action.usage_stats.success_rate * Math.log(action.usage_stats.total_executions + 1);
      console.log(`${i + 1}. ${action.name}`);
      console.log(`   Description: ${action.description}`);
      console.log(`   Type: ${action.type} | Category: ${action.category}`);
      console.log(`   Usage: ${action.usage_stats.total_executions:,} | Success: ${(action.usage_stats.success_rate * 100).toFixed(1)}%`);
      console.log(`   Relevance Score: ${score.toFixed(2)}\n`);
    });
    
    return sortedActions;
    
  } catch (error) {
    console.error('Failed to find actions:', error);
    throw error;
  }
}

// Usage examples
findActionsForUseCase('email');
findActionsForUseCase('data transformation');
```

### Pagination Examples

```typescript
async function getAllActionsPaginated() {
  const allActions = [];
  let cursor = null;
  
  do {
    const response = await client.actions.list({
      limit: 100,
      cursor: cursor,
      sort_by: 'created_at',
      sort_order: 'desc'
    });
    
    allActions.push(...response.actions);
    cursor = response.pagination.next_cursor;
    
    console.log(`Loaded ${allActions.length}/${response.pagination.total_count} actions`);
    
  } while (cursor);
  
  console.log(`Completed: ${allActions.length} total actions loaded`);
  return allActions;
}

async function getRecentlyCreatedActions(days = 30) {
  const cutoffDate = new Date(Date.now() - days * 24 * 60 * 60 * 1000);
  
  const response = await client.actions.list({
    sort_by: 'created_at',
    sort_order: 'desc',
    limit: 100
  });
  
  const recentActions = response.actions.filter(action => 
    new Date(action.created_at) >= cutoffDate
  );
  
  console.log(`Found ${recentActions.length} actions created in last ${days} days`);
  
  return recentActions;
}
```

## Action Categories

### Built-in Categories

## Performance Analysis

```typescript
async function analyzeActionPerformance() {
  const actions = await client.actions.list({
    limit: 100,
    sort_by: 'usage_count',
    sort_order: 'desc'
  });
  
  // Analyze performance metrics
  const performance = {
    highUsage: actions.actions.filter(a => a.usage_stats.total_executions > 1000),
    highReliability: actions.actions.filter(a => a.usage_stats.success_rate > 0.99),
    fastActions: actions.actions.filter(a => a.usage_stats.avg_duration_ms < 500),
    slowActions: actions.actions.filter(a => a.usage_stats.avg_duration_ms > 5000)
  };
  
  console.log('ðŸ“Š Action Performance Analysis:');
  console.log(`High Usage (>1000 executions): ${performance.highUsage.length}`);
  console.log(`High Reliability (>99% success): ${performance.highReliability.length}`);
  console.log(`Fast Actions (<500ms): ${performance.fastActions.length}`);
  console.log(`Slow Actions (>5s): ${performance.slowActions.length}`);
  
  // Performance recommendations
  const recommendations = [];
  
  if (performance.slowActions.length > 0) {
    recommendations.push(`Consider optimizing ${performance.slowActions.length} slow actions`);
  }
  
  const unreliableActions = actions.actions.filter(a => 
    a.usage_stats.success_rate < 0.9 && a.usage_stats.total_executions > 10
  );
  
  if (unreliableActions.length > 0) {
    recommendations.push(`Review ${unreliableActions.length} actions with low success rates`);
  }
  
  return { performance, recommendations };
}
```
