# Get Execution

Retrieve comprehensive information about a specific workflow execution, including current status, detailed results, step-by-step progress, performance metrics, and error details.

## Endpoint

<CodeGroup>
```http HTTP
GET /executions/{id}
```

```bash cURL
curl -X GET "https://tolstoy.getpullse.com/executions/exec_abc123def456" \
  -H "x-org-id: org_1234567890abcdef" \
  -H "x-user-id: user_abcdef1234567890"
```
</CodeGroup>

## Path Parameters

| Parameter | Type | Description |
|-----------|------|-------------|
| `id` | string | **Required.** The unique identifier of the execution to retrieve |

## Query Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `include_steps` | boolean | true | Include detailed step execution information |
| `include_logs` | boolean | false | Include execution logs in response |
| `include_profiling` | boolean | false | Include performance profiling data |
| `step_limit` | number | 50 | Maximum number of steps to return |

```typescript TypeScript/Node.js
import { TolstoyClient } from '@joosuhail/tolstoy-sdk';

const client = new TolstoyClient({
  orgId: 'org_1234567890abcdef',
  userId: 'user_abcdef1234567890'
});

async function getExecutionDetails(executionId: string) {
  try {
    const execution = await client.executions.get(executionId, {
      include_steps: true,
      include_profiling: true
    });
    
    console.log(`Execution: ${execution.id}`);
    console.log(`Flow: ${execution.flow_name} (v${execution.flow_version})`);
    console.log(`Status: ${execution.status}`);
    console.log(`Duration: ${execution.duration_ms || 'In progress'}ms`);
    
    if (execution.status === 'running') {
      console.log(`Progress: ${execution.progress_percentage}% (${execution.steps_completed}/${execution.total_steps} steps)`);
      console.log(`Current Step: ${execution.current_step}`);
      console.log(`Estimated Completion: ${execution.estimated_completion}`);
    }
    
    // Display step status
    console.log('\n📋 Step Status:');
    execution.steps.forEach((step, i) => {
      const statusIcon = {
        completed: '✅',
        running: '🏃',
        failed: '❌',
        skipped: '⏭️',
        pending: '⏳'
      }[step.status] || '❓';
      
      console.log(`${i + 1}. ${statusIcon} ${step.name} (${step.status})`);
      if (step.duration_ms) {
        console.log(`   Duration: ${step.duration_ms}ms`);
      }
      if (step.error) {
        console.log(`   Error: ${step.error.message}`);
      }
    });
    
    // Performance insights
    if (execution.performance) {
      console.log('\n⚡ Performance:');
      console.log(`  Network Time: ${execution.performance.network_time_ms}ms`);
      console.log(`  Processing Time: ${execution.performance.processing_time_ms}ms`);
      console.log(`  Retries: ${execution.performance.retries_total}`);
      
      if (execution.resource_usage) {
        console.log(`  Memory Peak: ${execution.resource_usage.memory_peak_mb}MB`);
      }
    }
    
    return execution;
    
  } catch (error) {
    console.error('Failed to get execution details:', error);
    throw error;
  }
}

// Usage
getExecutionDetails('exec_abc123def456');
```

```python Python
                duration = execution['duration_ms']
                print(f"🎉 Completed in {duration:,}ms")
                
                # Show outputs
                if execution.get('outputs'):
                    print("\n📤 Outputs:")
                    for key, value in execution['outputs'].items():
                        print(f"  {key}: {value}")
                
                # Performance summary
                if execution.get('performance'):
                    perf = execution['performance']
                    print(f"\n⚡ Performance Summary:")
                    print(f"  Total Duration: {perf['total_duration_ms']:,}ms")
                    print(f"  Network Time: {perf['network_time_ms']:,}ms")
                    print(f"  Processing Time: {perf['processing_time_ms']:,}ms")
                    print(f"  Retries: {perf['retries_total']}")
                
                break
                
            elif execution['status'] in ['failed', 'cancelled']:
                print(f"❌ Execution {execution['status']}")
```

### Advanced Analysis

```typescript
async function analyzeExecutionPerformance(executionId: string) {
  const execution = await client.executions.get(executionId, {
    include_steps: true,
    include_profiling: true
  });
  
  if (execution.status !== 'completed') {
    throw new Error('Can only analyze completed executions');
  }
  
  const analysis = {
    overall: {
      totalDuration: execution.duration_ms,
      successRate: execution.steps.filter(s => s.status === 'completed').length / execution.steps.length,
      retryRate: execution.steps.filter(s => s.execution_details?.attempts > 1).length / execution.steps.length
    },
    bottlenecks: [],
    recommendations: []
  };
  
  // Identify bottlenecks
  const avgDuration = execution.duration_ms / execution.steps.length;
  execution.steps.forEach(step => {
    if (step.duration_ms > avgDuration * 2) {
      analysis.bottlenecks.push({
        step: step.name,
        duration: step.duration_ms,
        impact: ((step.duration_ms / execution.duration_ms) * 100).toFixed(1) + '%'
      });
    }
  });
  
  // Generate recommendations
  if (analysis.overall.retryRate > 0.2) {
    analysis.recommendations.push('High retry rate detected - consider improving error handling');
  }
  
  if (execution.performance.network_time_ms > execution.performance.processing_time_ms * 5) {
    analysis.recommendations.push('Network time dominates execution - consider optimizing API calls');
  }
  
  if (analysis.bottlenecks.length > 0) {
    analysis.recommendations.push(`Optimize ${analysis.bottlenecks.length} bottleneck step(s)`);
  }
  
  return analysis;
}
```

## Execution Monitoring Patterns

### Real-time Monitoring

```typescript
class ExecutionMonitor {
  constructor(private client: TolstoyClient) {}
  
  async monitorWithCallback(
    executionId: string, 
    onUpdate: (execution: any) => void,
    options: { pollInterval?: number; timeout?: number } = {}
  ) {
    const { pollInterval = 2000, timeout = 300000 } = options;
    const startTime = Date.now();
    
    while (true) {
      const execution = await this.client.executions.get(executionId);
      onUpdate(execution);
      
      // Check if execution is complete
      if (['completed', 'failed', 'cancelled'].includes(execution.status)) {
        return execution;
      }
      
      // Check timeout
      if (Date.now() - startTime > timeout) {
        throw new Error('Monitoring timeout exceeded');
      }
      
      // Wait before next poll
      await new Promise(resolve => setTimeout(resolve, pollInterval));
    }
  }
  
  async waitForCompletion(
    executionId: string, 
    options: { timeout?: number; pollInterval?: number } = {}
  ) {
    return this.monitorWithCallback(executionId, (execution) => {
      if (execution.status === 'running') {
        console.log(`Progress: ${execution.progress_percentage}% - ${execution.current_step}`);
      }
    }, options);
  }
}

// Usage
const monitor = new ExecutionMonitor(client);

monitor.waitForCompletion('exec_abc123def456', {
  timeout: 600000, // 10 minutes
  pollInterval: 3000 // 3 seconds
}).then(result => {
  console.log('Execution completed:', result.status);
}).catch(error => {
  console.error('Monitoring failed:', error);
});
```

## Error Analysis

```typescript
function analyzeExecutionErrors(execution: any) {
  if (execution.status !== 'failed') {
    return { hasErrors: false };
  }
  
  const failedSteps = execution.steps.filter(s => s.status === 'failed');
  const errorAnalysis = {
    hasErrors: true,
    totalErrors: failedSteps.length,
    errorTypes: {},
    retriableErrors: 0,
    permanentErrors: 0,
    recommendations: []
  };
  
  failedSteps.forEach(step => {
    const errorCode = step.error?.code || 'UNKNOWN_ERROR';
    errorAnalysis.errorTypes[errorCode] = (errorAnalysis.errorTypes[errorCode] || 0) + 1;
    
    // Classify error type
    const retriableCodes = ['TIMEOUT', 'RATE_LIMIT', 'NETWORK_ERROR', 'TEMPORARY_FAILURE'];
    if (retriableCodes.includes(errorCode)) {
      errorAnalysis.retriableErrors++;
    } else {
      errorAnalysis.permanentErrors++;
    }
  });
  
  // Generate recommendations
  if (errorAnalysis.retriableErrors > 0) {
    errorAnalysis.recommendations.push('Consider increasing retry limits for transient failures');
  }
  
  if (errorAnalysis.errorTypes['TIMEOUT'] > 0) {
    errorAnalysis.recommendations.push('Increase timeout values for slow operations');
  }
  
  if (errorAnalysis.errorTypes['RATE_LIMIT'] > 0) {
    errorAnalysis.recommendations.push('Implement exponential backoff or rate limiting');
  }
  
  return errorAnalysis;
}
```
